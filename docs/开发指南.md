### 3.1 jt809-core 核心模块

#### 3.1.1 消息定义规范

**步骤1**: 定义消息体
```java
package org.iot.v.jt809.core.message.upstream;

import org.iot.v.jt809.core.message.base.BaseMessage;
import org.iot.v.jt809.core.message.base.MessageBody;
import lombok.Data;
import lombok.EqualsAndHashCode;

/**
 * 上行连接请求消息 (消息ID: 0x1001)
 */
@Data
@EqualsAndHashCode(callSuper = true)
public class UpConnectReq extends BaseMessage {
    
    public UpConnectReq() {
        getHead().setMsgId(0x1001);
    }
    
    @Data
    @EqualsAndHashCode(callSuper = true)
    public static class Body extends MessageBody {
        private long userId;          // 用户ID (4字节)
        private String password;       // 密码 (8字节)
        private String downLinkIp;     // 下级平台IP (32字节)
        private int downLinkPort;      // 下级平台端口 (4字节)
        
        @Override
        public byte[] encode() {
            ByteBuf buffer = Unpooled.buffer(48);
            buffer.writeInt((int) userId);
            buffer.writeBytes(password.getBytes(), 0, 8);
            buffer.writeBytes(downLinkIp.getBytes(), 0, 32);
            buffer.writeInt(downLinkPort);
            return buffer.array();
        }
        
        @Override
        public void decode(byte[] data) {
            ByteBuf buffer = Unpooled.wrappedBuffer(data);
            this.userId = buffer.readUnsignedInt();
            byte[] pwdBytes = new byte[8];
            buffer.readBytes(pwdBytes);
            this.password = new String(pwdBytes).trim();
            byte[] ipBytes = new byte[32];
            buffer.readBytes(ipBytes);
            this.downLinkIp = new String(ipBytes).trim();
            this.downLinkPort = buffer.readInt();
        }
    }
}
```

#### 3.1.3 工具类开发

**CRC校验工具**:
```java
public class CRCUtil {
    /**
     * 计算CRC-CCITT校验码
     * 多项式: 0x1021
     * 初始值: 0xFFFF
     */
    public static int calculate(byte[] data) {
        int crc = 0xFFFF;
        for (byte b : data) {
            crc ^= (b << 8);
            for (int i = 0; i < 8; i++) {
                if ((crc & 0x8000) != 0) {
                    crc = (crc << 1) ^ 0x1021;
                } else {
                    crc <<= 1;
                }
            }
        }
        return crc & 0xFFFF;
    }
    
    /**
     * 校验数据
     */
    public static boolean verify(byte[] data, int expectedCrc) {
        return calculate(data) == expectedCrc;
    }
}
```

```java
public class ByteUtil {
    /**
     * byte数组转十六进制字符串
     */
    public static String bytesToHex(byte[] bytes) {
        StringBuilder sb = new StringBuilder();
        for (byte b : bytes) {
            sb.append(String.format("%02X ", b));
        }
        return sb.toString();
    }
    
    /**
     * 十六进制字符串转byte数组
     */
    public static byte[] hexToBytes(String hex) {
        String[] hexArray = hex.split(" ");
        byte[] bytes = new byte[hexArray.length];
        for (int i = 0; i < hexArray.length; i++) {
            bytes[i] = (byte) Integer.parseInt(hexArray[i], 16);
        }
        return bytes;
    }
}
```

### 3.2 jt809-server 服务端模块

#### 3.2.1 服务端主类开发

```java
@Slf4j
public class JT809Server {
    private final ServerProperties properties;
    private final HandlerChain handlerChain;
    
    private EventLoopGroup bossGroup;
    private EventLoopGroup workerGroup;
    private Channel serverChannel;
    
    public JT809Server(ServerProperties properties, HandlerChain handlerChain) {
        this.properties = properties;
        this.handlerChain = handlerChain;
    }
    
    /**
     * 启动服务端
     */
    public void start() {
        bossGroup = new NioEventLoopGroup(properties.getBossThreads());
        workerGroup = new NioEventLoopGroup(properties.getWorkerThreads());
        
        try {
            ServerBootstrap bootstrap = new ServerBootstrap();
            bootstrap.group(bossGroup, workerGroup)
                .channel(NioServerSocketChannel.class)
                .option(ChannelOption.SO_BACKLOG, 128)
                .childOption(ChannelOption.SO_KEEPALIVE, true)
                .childOption(ChannelOption.TCP_NODELAY, true)
                .childHandler(new ChannelInitializer<SocketChannel>() {
                    @Override
                    protected void initChannel(SocketChannel ch) {
                        ChannelPipeline pipeline = ch.pipeline();
                        
                        // 添加编解码器
                        pipeline.addLast("decoder", new JT809Decoder());
                        pipeline.addLast("encoder", new JT809Encoder());
                        
                        // 添加业务处理器
                        pipeline.addLast("serverHandler", 
                            new ServerHandler(handlerChain));
                        
                        // 添加空闲检测
                        pipeline.addLast("idleState", 
                            new IdleStateHandler(180, 0, 0, TimeUnit.SECONDS));
                        pipeline.addLast("idleHandler", 
                            new ServerIdleHandler());
                    }
                });
            
            ChannelFuture future = bootstrap.bind(properties.getPort()).sync();
            serverChannel = future.channel();
            
            log.info("JT809 Server started on port: {}", properties.getPort());
            
        } catch (InterruptedException e) {
            log.error("Server start failed", e);
            shutdown();
            Thread.currentThread().interrupt();
        }
    }
    
    /**
     * 停止服务端
     */
    public void stop() {
        log.info("Stopping JT809 Server...");
        
        if (serverChannel != null) {
            serverChannel.close();
        }
        
        shutdown();
        
        log.info("JT809 Server stopped");
    }
    
    private void shutdown() {
        if (bossGroup != null) {
            bossGroup.shutdownGracefully();
        }
        if (workerGroup != null) {
            workerGroup.shutdownGracefully();
        }
    }
}
```

#### 3.2.2 服务端处理器

```java
@Slf4j
public class ServerHandler extends SimpleChannelInboundHandler<BaseMessage> {
    private final HandlerChain handlerChain;
    
    public ServerHandler(HandlerChain handlerChain) {
        this.handlerChain = handlerChain;
    }
    
    @Override
    protected void channelRead0(ChannelHandlerContext ctx, BaseMessage msg) {
        log.debug("Received message: {}", msg.getHead().getMsgId());
        
        // 创建会话（如果不存在）
        Session session = SessionManager.getInstance()
            .getSession(ctx.channel().id().asLongText());
        if (session == null) {
            session = new Session(ctx.channel());
            SessionManager.getInstance().addSession(session);
        }
        
        // 创建消息上下文
        MessageContext context = new MessageContext();
        context.setSession(session);
        context.setRequest(msg);
        
        // 通过处理器链处理消息
        handlerChain.process(context, msg);
    }
    
    @Override
    public void channelActive(ChannelHandlerContext ctx) {
        log.info("New connection from: {}", ctx.channel().remoteAddress());
    }
    
    @Override
    public void channelInactive(ChannelHandlerContext ctx) {
        log.info("Connection closed: {}", ctx.channel().remoteAddress());
        
        // 移除会话
        Session session = SessionManager.getInstance()
            .getSession(ctx.channel().id().asLongText());
        if (session != null) {
            SessionManager.getInstance().removeSession(session.getSessionId());
        }
    }
    
    @Override
    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
        log.error("Exception caught", cause);
        ctx.close();
    }
}
```

### 3.3 jt809-client 客户端模块

#### 3.3.1 客户端主类开发

```java
@Slf4j
public class JT809Client {
    private final ClientProperties properties;
    private final HandlerChain handlerChain;
    
    private EventLoopGroup workerGroup;
    private Channel clientChannel;
    private ScheduledExecutorService scheduler;
    
    public JT809Client(ClientProperties properties, HandlerChain handlerChain) {
        this.properties = properties;
        this.handlerChain = handlerChain;
        this.scheduler = Executors.newSingleThreadScheduledExecutor();
    }
    
    /**
     * 连接服务器
     */
    public void connect() {
        workerGroup = new NioEventLoopGroup();
        
        try {
            Bootstrap bootstrap = new Bootstrap();
            bootstrap.group(workerGroup)
                .channel(NioSocketChannel.class)
                .option(ChannelOption.TCP_NODELAY, true)
                .option(ChannelOption.SO_KEEPALIVE, true)
                .option(ChannelOption.CONNECT_TIMEOUT_MILLIS, 10000)
                .handler(new ChannelInitializer<SocketChannel>() {
                    @Override
                    protected void initChannel(SocketChannel ch) {
                        ChannelPipeline pipeline = ch.pipeline();
                        
                        // 编解码器
                        pipeline.addLast("decoder", new JT809Decoder());
                        pipeline.addLast("encoder", new JT809Encoder());
                        
                        // 业务处理器
                        pipeline.addLast("clientHandler", 
                            new ClientHandler(handlerChain));
                        
                        // 重连处理器
                        pipeline.addLast("reconnectHandler", 
                            new ReconnectHandler(properties.getReconnect()));
                        
                        // 空闲检测
                        pipeline.addLast("idleState", 
                            new IdleStateHandler(0, 60, 0, TimeUnit.SECONDS));
                        pipeline.addLast("heartbeatHandler", 
                            new HeartbeatHandler());
                    }
                });
            
            ChannelFuture future = bootstrap.connect(
                properties.getServerIp(), 
                properties.getServerPort()
            );
            
            future.addListener((ChannelFutureListener) f -> {
                if (f.isSuccess()) {
                    log.info("Connected to server: {}:{}", 
                        properties.getServerIp(), properties.getServerPort());
                    clientChannel = f.channel();
                    
                    // 发送登录请求
                    login();
                    
                    // 启动心跳任务
                    startHeartbeat();
                } else {
                    log.error("Failed to connect to server");
                    if (properties.getReconnect().isEnabled()) {
                        scheduleReconnect();
                    }
                }
            });
            
        } catch (Exception e) {
            log.error("Connect failed", e);
            if (properties.getReconnect().isEnabled()) {
                scheduleReconnect();
            }
        }
    }
    
    /**
     * 断开连接
     */
    public void disconnect() {
        log.info("Disconnecting from server...");
        
        // 停止心跳
        scheduler.shutdown();
        
        // 发送断开请求
        if (clientChannel != null && clientChannel.isActive()) {
            UpDisconnectReq disconnectReq = new UpDisconnectReq();
            clientChannel.writeAndFlush(disconnectReq);
        }
        
        // 关闭通道
        if (clientChannel != null) {
            clientChannel.close();
        }
        
        // 关闭线程组
        if (workerGroup != null) {
            workerGroup.shutdownGracefully();
        }
        
        log.info("Disconnected");
    }
    
    /**
     * 发送消息
     */
    public void send(BaseMessage message) {
        if (clientChannel != null && clientChannel.isActive()) {
            clientChannel.writeAndFlush(message);
        } else {
            log.warn("Channel is not active, cannot send message");
        }
    }
    
    /**
     * 登录
     */
    private void login() {
        UpConnectReq loginReq = new UpConnectReq();
        loginReq.setBody(new UpConnectReq.Body());
        loginReq.getBody().setUserId(properties.getPlatformId());
        loginReq.getBody().setPassword(properties.getPassword());
        // ... 设置其他字段
        
        send(loginReq);
    }
    
    /**
     * 启动心跳
     */
    private void startHeartbeat() {
        if (properties.getHeartbeat().isEnabled()) {
            scheduler.scheduleAtFixedRate(() -> {
                if (clientChannel != null && clientChannel.isActive()) {
                    UpLinkTestReq heartbeat = new UpLinkTestReq();
                    send(heartbeat);
                }
            }, properties.getHeartbeat().getInterval(), 
               properties.getHeartbeat().getInterval(), 
               TimeUnit.MILLISECONDS);
        }
    }
    
    /**
     * 调度重连
     */
    private void scheduleReconnect() {
        scheduler.schedule(() -> {
            log.info("Reconnecting...");
            connect();
        }, properties.getReconnect().getInterval(), TimeUnit.MILLISECONDS);
    }
}
```

### 3.4 jt809-handler 消息处理器模块

#### 3.4.1 处理器接口实现

```java
/**
 * 抽象消息处理器，提供基础功能
 */
@Slf4j
public abstract class AbstractMessageHandler implements MessageHandler {
    
    @Override
    public boolean handle(MessageContext context, BaseMessage message) {
        try {
            // 前置检查
            if (!preHandle(context, message)) {
                return false;
            }
            
            // 实际处理
            doHandle(context, message);
            
            // 后置处理
            postHandle(context, message);
            
            return true;
            
        } catch (Exception e) {
            log.error("Handle message failed: {}", message.getHead().getMsgId(), e);
            onError(context, message, e);
            return false;
        }
    }
    
    /**
     * 前置处理
     */
    protected boolean preHandle(MessageContext context, BaseMessage message) {
        return true;
    }
    
    /**
     * 实际处理逻辑（子类实现）
     */
    protected abstract void doHandle(MessageContext context, BaseMessage message);
    
    /**
     * 后置处理
     */
    protected void postHandle(MessageContext context, BaseMessage message) {
        // 默认空实现
    }
    
    /**
     * 异常处理
     */
    protected void onError(MessageContext context, BaseMessage message, Exception e) {
        // 默认空实现
    }
}
```

#### 3.4.2 具体处理器示例

```java
/**
 * 登录处理器
 */
@JT809Handler(messageTypes = {0x1001}, order = 1)
public class LoginHandler extends AbstractMessageHandler {
    
    @Autowired
    private AuthService authService;
    
    @Override
    protected void doHandle(MessageContext context, BaseMessage message) {
        UpConnectReq loginReq = (UpConnectReq) message;
        UpConnectReq.Body body = loginReq.getBody();
        
        // 验证登录
        boolean authenticated = authService.authenticate(
            body.getUserId(), 
            body.getPassword()
        );
        
        // 创建响应
        UpConnectResp response = new UpConnectResp();
        UpConnectResp.Body respBody = new UpConnectResp.Body();
        
        if (authenticated) {
            // 更新会话信息
            context.getSession().setPlatformId(body.getUserId());
            context.getSession().setState(SessionState.AUTHENTICATED);
            
            respBody.setResult(0); // 成功
            respBody.setVerifyCode(generateVerifyCode());
            
            log.info("Platform login success: {}", body.getUserId());
        } else {
            respBody.setResult(1); // 失败
            log.warn("Platform login failed: {}", body.getUserId());
        }
        
        response.setBody(respBody);
        
        // 发送响应
        context.sendResponse(response);
    }
    
    private long generateVerifyCode() {
        return System.currentTimeMillis();
    }
    
    @Override
    public int[] supportedMessageTypes() {
        return new int[]{0x1001};
    }
}
```

### 3.5 jt809-spring-boot-starter 自动配置模块

#### 3.5.1 自动配置类

```java
/**
 * JT809自动配置
 */
@Configuration
@EnableConfigurationProperties(JT809Properties.class)
@ConditionalOnProperty(prefix = "jt809", name = "enabled", 
    havingValue = "true", matchIfMissing = true)
public class JT809AutoConfiguration {
    
    @Bean
    @ConditionalOnMissingBean
    public SessionManager sessionManager() {
        return SessionManager.getInstance();
    }
    
    @Bean
    @ConditionalOnMissingBean
    public HandlerChain handlerChain() {
        return new HandlerChain();
    }
    
    @Bean
    public MessageHandlerPostProcessor messageHandlerPostProcessor(
            HandlerChain handlerChain) {
        return new MessageHandlerPostProcessor(handlerChain);
    }
}

/**
 * 处理器后置处理器，自动注册带有@JT809Handler注解的处理器
 */
public class MessageHandlerPostProcessor 
        implements BeanPostProcessor, ApplicationContextAware {
    
    private final HandlerChain handlerChain;
    private ApplicationContext applicationContext;
    
    public MessageHandlerPostProcessor(HandlerChain handlerChain) {
        this.handlerChain = handlerChain;
    }
    
    @Override
    public Object postProcessAfterInitialization(Object bean, String beanName) {
        if (bean instanceof MessageHandler) {
            MessageHandler handler = (MessageHandler) bean;
            
            // 检查是否有@JT809Handler注解
            JT809Handler annotation = bean.getClass()
                .getAnnotation(JT809Handler.class);
            
            if (annotation != null) {
                handlerChain.addHandler(handler);
                log.info("Registered JT809 handler: {}", 
                    bean.getClass().getSimpleName());
            }
        }
        return bean;
    }
    
    @Override
    public void setApplicationContext(ApplicationContext applicationContext) 
            throws BeansException {
        this.applicationContext = applicationContext;
    }
}
```
