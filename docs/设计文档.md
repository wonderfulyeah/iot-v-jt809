# JT809 数据级联 Spring Starter 设计文档

## 1. 项目概述

### 1.1 项目目标
实现一个基于Spring Boot的JT809协议数据级联框架，支持：
- 作为上级平台接收下级平台的数据上报
- 作为下级平台向上级平台注册并上报数据
- 灵活的消息处理机制，支持自定义业务逻辑
- 开箱即用的Spring Boot Starter集成

### 1.2 技术栈
- **Java**: 8+
- **Spring Boot**: 2.6.3
- **Netty**: 4.1.72.Final (核心网络通信框架)
- **Spring Framework**: 5.3.15 (随Spring Boot 2.6.3自带)
- **日志**: SLF4J + Logback (Spring Boot默认)
- **工具**: Lombok, Hutool (可选)

### 1.3 核心特性
- **Netty高性能网络层**: 基于Netty的异步事件驱动架构
- **零拷贝优化**: 使用Netty ByteBuf减少内存拷贝
- **Reactor响应式**: 支持响应式编程模型
- **连接池管理**: 高效的连接管理和复用
- **优雅停机**: 支持Spring生命周期管理

#### 3.2.1 消息编码器（Netty实现）
```java
/**
 * JT809消息编码器 - 基于Netty的MessageToByteEncoder
 * 使用ByteBuf实现零拷贝优化
 */
@Slf4j
public class JT809Encoder extends MessageToByteEncoder<BaseMessage> {
    
    private final EscapeHandler escapeHandler = new EscapeHandler();
    
    @Override
    protected void encode(ChannelHandlerContext ctx, BaseMessage msg, ByteBuf out) {
        try {
            // 1. 编码消息体
            ByteBuf bodyBuf = encodeBody(msg.getBody());
            
            // 2. 编码消息头
            ByteBuf headBuf = encodeHead(msg.getHead(), bodyBuf.readableBytes());
            
            // 3. 计算CRC校验（未转义前的数据）
            ByteBuf crcBuf = Unpooled.wrappedBuffer(headBuf, bodyBuf);
            int crc = CRCUtil.calculate(crcBuf);
            
            // 4. 转义处理（标识符之外的完整消息）
            ByteBuf messageBuf = Unpooled.buffer();
            messageBuf.writeBytes(headBuf);
            messageBuf.writeBytes(bodyBuf);
            messageBuf.writeShort(crc);
            
            ByteBuf escapedBuf = escapeHandler.escape(messageBuf);
            
            // 5. 组装完整消息
            // 标识位(0x5b) + 转义后的消息 + 标识位(0x5d)
            out.writeByte(JT809Constant.FLAG_START);
            out.writeBytes(escapedBuf);
            out.writeByte(JT809Constant.FLAG_END);
            
            // 6. 释放临时ByteBuf
            ReferenceCountUtil.release(bodyBuf);
            ReferenceCountUtil.release(headBuf);
            ReferenceCountUtil.release(crcBuf);
            ReferenceCountUtil.release(messageBuf);
            ReferenceCountUtil.release(escapedBuf);
            
        } catch (Exception e) {
            log.error("Encode message failed", e);
            throw new CodecException("Encode message failed", e);
        }
    }
    
    /**
     * 编码消息体
     */
    private ByteBuf encodeBody(MessageBody body) {
        ByteBuf buffer = Unpooled.buffer(256);
        byte[] bodyBytes = body.encode();
        buffer.writeBytes(bodyBytes);
        return buffer;
    }
    
    /**
     * 编码消息头
     */
    private ByteBuf encodeHead(MessageHead head, int bodyLength) {
        ByteBuf buffer = Unpooled.buffer(22);
        
        // 消息长度 = 消息头(22) + 消息体长度 + CRC(2)
        buffer.writeInt(22 + bodyLength + 2);
        
        // 消息流水号
        buffer.writeInt(head.getMsgSn());
        
        // 消息ID
        buffer.writeShort(head.getMsgId());
        
        // 下级平台接入码 (BCD码)
        buffer.writeBytes(BCDUtil.encode(head.getPlatformId(), 8));
        
        // 协议版本
        buffer.writeByte(head.getVersion().getMajor());
        buffer.writeByte(head.getVersion().getMinor());
        
        // 上级平台接入码
        buffer.writeInt((int) head.getSuperPlatformId());
        
        // 加密方式
        buffer.writeByte(head.getEncrypt());
        
        // 加密密钥
        buffer.writeInt((int) head.getEncryptKey());
        
        return buffer;
    }
}
```

#### 3.2.2 消息解码器（Netty实现）
```java
/**
 * JT809消息解码器 - 基于Netty的ByteToMessageDecoder
 * 支持粘包拆包处理
 */
@Slf4j
public class JT809Decoder extends ByteToMessageDecoder {
    
    private final EscapeHandler escapeHandler = new EscapeHandler();
    
    // 最小消息长度：起始符(1) + 消息头(22) + CRC(2) + 结束符(1) = 26字节
    private static final int MIN_MESSAGE_LENGTH = 26;
    
    @Override
    protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) {
        try {
            // 1. 检查可读字节数
            if (in.readableBytes() < MIN_MESSAGE_LENGTH) {
                return;
            }
            
            // 2. 标记读取位置（用于半包处理）
            in.markReaderIndex();
            
            // 3. 查找消息起始标识 0x5b
            int startIndex = findStartFlag(in);
            if (startIndex == -1) {
                // 没有找到起始标识，丢弃无效数据
                in.skipBytes(in.readableBytes());
                return;
            }
            
            // 4. 查找消息结束标识 0x5d
            int endIndex = findEndFlag(in, startIndex);
            if (endIndex == -1) {
                // 没有找到结束标识，可能是半包，重置读取位置等待更多数据
                in.resetReaderIndex();
                return;
            }
            
            // 5. 读取完整消息（去除起始和结束标识）
            int messageLength = endIndex - startIndex - 1;
            ByteBuf escapedMessage = in.readSlice(messageLength);
            
            // 6. 反转义处理
            ByteBuf originalMessage = escapeHandler.unescape(escapedMessage);
            
            // 7. 读取消息长度并验证
            int declaredLength = originalMessage.readInt();
            int actualLength = originalMessage.readableBytes() + 4; // 加上已读的长度字段
            
            if (declaredLength != actualLength) {
                log.warn("Message length mismatch: declared={}, actual={}", 
                    declaredLength, actualLength);
                ReferenceCountUtil.release(originalMessage);
                return;
            }
            
            // 8. CRC校验
            ByteBuf messageData = originalMessage.slice(0, originalMessage.readableBytes() - 2);
            int calculatedCrc = CRCUtil.calculate(messageData);
            int receivedCrc = originalMessage.getUnsignedShort(
                originalMessage.readableBytes() - 2);
            
            if (calculatedCrc != receivedCrc) {
                log.warn("CRC check failed: calculated={}, received={}", 
                    calculatedCrc, receivedCrc);
                ReferenceCountUtil.release(originalMessage);
                return;
            }
            
            // 9. 解码消息头
            MessageHead head = decodeHead(originalMessage);
            
            // 10. 根据消息ID创建对应的消息对象
            BaseMessage message = createMessage(head.getMsgId());
            message.setHead(head);
            
            // 11. 解码消息体
            int bodyLength = declaredLength - 22 - 2; // 减去消息头和CRC长度
            byte[] bodyBytes = new byte[bodyLength];
            originalMessage.readBytes(bodyBytes);
            message.getBody().decode(bodyBytes);
            
            // 12. 添加到输出列表
            out.add(message);
            
            // 13. 跳过结束标识
            in.skipBytes(1);
            
            // 14. 释放临时ByteBuf
            ReferenceCountUtil.release(originalMessage);
            
        } catch (Exception e) {
            log.error("Decode message failed", e);
            // 异常时跳过当前字节，继续尝试解析
            in.skipBytes(1);
        }
    }
    
    /**
     * 查找起始标识
     */
    private int findStartFlag(ByteBuf buf) {
        for (int i = buf.readerIndex(); i < buf.writerIndex(); i++) {
            if (buf.getByte(i) == JT809Constant.FLAG_START) {
                return i;
            }
        }
        return -1;
    }
    
    /**
     * 查找结束标识
     */
    private int findEndFlag(ByteBuf buf, int startIndex) {
        for (int i = startIndex + 1; i < buf.writerIndex(); i++) {
            if (buf.getByte(i) == JT809Constant.FLAG_END) {
                return i;
            }
        }
        return -1;
    }
    
    /**
     * 解码消息头
     */
    private MessageHead decodeHead(ByteBuf buf) {
        MessageHead head = new MessageHead();
        
        // 消息长度（已读取）
        // 消息流水号
        head.setMsgSn(buf.readInt());
        
        // 消息ID
        head.setMsgId(buf.readUnsignedShort());
        
        // 下级平台接入码 (BCD码)
        byte[] platformIdBytes = new byte[8];
        buf.readBytes(platformIdBytes);
        head.setPlatformId(BCDUtil.decodeToLong(platformIdBytes));
        
        // 协议版本
        head.setVersion(new ProtocolVersion(buf.readByte(), buf.readByte()));
        
        // 上级平台接入码
        head.setSuperPlatformId(buf.readUnsignedInt());
        
        // 加密方式
        head.setEncrypt(buf.readByte());
        
        // 加密密钥
        head.setEncryptKey(buf.readUnsignedInt());
        
        return head;
    }
    
    /**
     * 根据消息ID创建消息对象
     */
    private BaseMessage createMessage(int msgId) {
        Class<? extends BaseMessage> msgClass = 
            MessageTypeRegistry.getMessageClass(msgId);
        
        if (msgClass == null) {
            throw new CodecException("Unknown message type: " + msgId);
        }
        
        try {
            return msgClass.newInstance();
        } catch (Exception e) {
            throw new CodecException("Create message instance failed", e);
        }
    }
}
```

```java
public class SessionManager {
    private static final SessionManager INSTANCE = new SessionManager();
    private final Map<String, Session> sessions = new ConcurrentHashMap<>();
    private final Map<Long, Session> platformSessions = new ConcurrentHashMap<>();

    public void addSession(Session session) {
        sessions.put(session.getSessionId(), session);
        platformSessions.put(session.getPlatformId(), session);
    }
    
    // 移除会话
    public void removeSession(String sessionId) {
        Session session = sessions.remove(sessionId);
        if (session != null) {
            platformSessions.remove(session.getPlatformId());
        }
    }

        return platformSessions.get(platformId);
    }
    
    // 广播消息
    public void broadcast(BaseMessage message) {
        sessions.values().forEach(s -> s.send(message));
    }
}
```

### 3.4 服务端设计

#### 3.4.1 服务端核心类（Netty实现）
```java
/**
 * JT809服务端 - 基于Netty实现
 * 支持高性能、高并发连接处理
 */
@Slf4j
public class JT809Server implements SmartLifecycle {
    
    private final JT809Properties.ServerConfig config;
    private final HandlerChain handlerChain;
    private final SessionManager sessionManager;
    
    // Netty组件
    private EventLoopGroup bossGroup;
    private EventLoopGroup workerGroup;
    private Channel serverChannel;
    
    // 业务线程池
    private EventLoopGroup businessGroup;
    
    // 生命周期状态
    private volatile boolean running = false;
    
    public JT809Server(JT809Properties.ServerConfig config, 
                       HandlerChain handlerChain,
                       SessionManager sessionManager) {
        this.config = config;
        this.handlerChain = handlerChain;
        this.sessionManager = sessionManager;
    }
    
    /**
     * 启动服务端
     */
    @Override
    public void start() {
        if (running) {
            return;
        }
        
        log.info("Starting JT809 Server on port: {}", config.getPort());
        
        try {
            // 1. 创建EventLoopGroup
            bossGroup = new NioEventLoopGroup(config.getBossThreads(), 
                new DefaultThreadFactory("jt809-server-boss"));
            
            workerGroup = new NioEventLoopGroup(config.getWorkerThreads(),
                new DefaultThreadFactory("jt809-server-worker"));
            
            businessGroup = new NioEventLoopGroup(config.getBusinessThreads(),
                new DefaultThreadFactory("jt809-server-business"));
            
            // 2. 配置ServerBootstrap
            ServerBootstrap bootstrap = new ServerBootstrap();
            bootstrap.group(bossGroup, workerGroup)
                .channel(NioServerSocketChannel.class)
                .option(ChannelOption.SO_BACKLOG, 1024)
                .option(ChannelOption.SO_REUSEADDR, true)
                .childOption(ChannelOption.TCP_NODELAY, true)
                .childOption(ChannelOption.SO_KEEPALIVE, true)
                .childOption(ChannelOption.SO_RCVBUF, 32 * 1024)
                .childOption(ChannelOption.SO_SNDBUF, 32 * 1024)
                .childOption(ChannelOption.ALLOCATOR, PooledByteBufAllocator.DEFAULT)
                .childHandler(new ChannelInitializer<SocketChannel>() {
                    @Override
                    protected void initChannel(SocketChannel ch) {
                        initChannelPipeline(ch);
                    }
                });
            
            // 3. 绑定端口
            ChannelFuture future = bootstrap.bind(config.getPort()).sync();
            serverChannel = future.channel();
            
            running = true;
            log.info("JT809 Server started successfully on port: {}", config.getPort());
            
        } catch (InterruptedException e) {
            log.error("Start JT809 Server failed", e);
            shutdown();
            Thread.currentThread().interrupt();
            throw new RuntimeException("Start JT809 Server failed", e);
        }
    }
    
    /**
     * 初始化Channel Pipeline
     */
    private void initChannelPipeline(SocketChannel ch) {
        ChannelPipeline pipeline = ch.pipeline();
        
        // 1. 空闲检测（180秒无读操作触发）
        pipeline.addLast("idleState", 
            new IdleStateHandler(180, 0, 0, TimeUnit.SECONDS));
        
        // 2. 编解码器
        pipeline.addLast("decoder", new JT809Decoder());
        pipeline.addLast("encoder", new JT809Encoder());
        
        // 3. 业务处理器（使用业务线程池）
        pipeline.addLast(businessGroup, "serverHandler", 
            new ServerHandler(handlerChain, sessionManager));
        
        // 4. 空闲处理
        pipeline.addLast("idleHandler", new ServerIdleHandler());
        
        // 5. 异常处理
        pipeline.addLast("exceptionHandler", new ServerExceptionHandler());
        
        // 6. 日志处理器（可选）
        if (config.isEnableLog()) {
            pipeline.addLast("logging", 
                new LoggingHandler(LogLevel.DEBUG));
        }
    }
    
    /**
     * 停止服务端
     */
    @Override
    public void stop() {
        if (!running) {
            return;
        }
        
        log.info("Stopping JT809 Server...");
        
        // 1. 关闭服务端Channel
        if (serverChannel != null) {
            serverChannel.close().awaitUninterruptibly();
        }
        
        // 2. 优雅关闭EventLoopGroup
        shutdown();
        
        running = false;
        log.info("JT809 Server stopped");
    }
    
    /**
     * 优雅关闭
     */
    private void shutdown() {
        // 优雅关闭，等待任务完成
        if (bossGroup != null) {
            bossGroup.shutdownGracefully(100, 300, TimeUnit.MILLISECONDS);
        }
        if (workerGroup != null) {
            workerGroup.shutdownGracefully(100, 300, TimeUnit.MILLISECONDS);
        }
```

#### 3.5.1 客户端核心类（Netty实现）
```java
/**
 * JT809客户端 - 基于Netty实现
 * 支持断线重连、心跳保活
 */
@Slf4j
public class JT809Client implements SmartLifecycle {
    
    private final JT809Properties.ClientConfig config;
    private final HandlerChain handlerChain;
    private final SessionManager sessionManager;
    
    // Netty组件
    private EventLoopGroup workerGroup;
    private EventLoopGroup businessGroup;
    private Channel clientChannel;
    private Bootstrap bootstrap;
    
    // 定时任务
    private ScheduledFuture<?> heartbeatFuture;
    private ScheduledFuture<?> reconnectFuture;
    
    // 生命周期状态
    private volatile boolean running = false;
    private volatile boolean loginSuccess = false;
    
    public JT809Client(JT809Properties.ClientConfig config,
                       HandlerChain handlerChain,
                       SessionManager sessionManager) {
        this.config = config;
        this.handlerChain = handlerChain;
        this.sessionManager = sessionManager;
    }
    
    /**
     * 启动客户端
     */
    @Override
    public void start() {
        if (running) {
            return;
        }
        
        log.info("Starting JT809 Client, target: {}:{}", 
            config.getServerIp(), config.getServerPort());
        
        try {
            // 1. 创建EventLoopGroup
            workerGroup = new NioEventLoopGroup(config.getWorkerThreads(),
                new DefaultThreadFactory("jt809-client-worker"));
            
            businessGroup = new NioEventLoopGroup(config.getBusinessThreads(),
                new DefaultThreadFactory("jt809-client-business"));
            
            // 2. 配置Bootstrap
            bootstrap = new Bootstrap();
            bootstrap.group(workerGroup)
                .channel(NioSocketChannel.class)
                .option(ChannelOption.TCP_NODELAY, true)
                .option(ChannelOption.SO_KEEPALIVE, true)
                .option(ChannelOption.CONNECT_TIMEOUT_MILLIS, 
                    config.getConnectTimeout())
                .option(ChannelOption.SO_RCVBUF, 32 * 1024)
                .option(ChannelOption.SO_SNDBUF, 32 * 1024)
                .option(ChannelOption.ALLOCATOR, PooledByteBufAllocator.DEFAULT)
                .handler(new ChannelInitializer<SocketChannel>() {
                    @Override
                    protected void initChannel(SocketChannel ch) {
                        initChannelPipeline(ch);
                    }
                });
            
            // 3. 连接服务器
            connect();
            
            running = true;
            log.info("JT809 Client started successfully");
            
        } catch (Exception e) {
            log.error("Start JT809 Client failed", e);
            shutdown();
            throw new RuntimeException("Start JT809 Client failed", e);
        }
    }
    
    /**
     * 初始化Channel Pipeline
     */
    private void initChannelPipeline(SocketChannel ch) {
        ChannelPipeline pipeline = ch.pipeline();
        
        // 1. 空闲检测（读空闲60秒，写空闲60秒）
        pipeline.addLast("idleState",
            new IdleStateHandler(0, config.getHeartbeat().getInterval() / 1000, 0, 
                TimeUnit.MILLISECONDS));
        
        // 2. 编解码器
        pipeline.addLast("decoder", new JT809Decoder());
        pipeline.addLast("encoder", new JT809Encoder());
        
        // 3. 业务处理器（使用业务线程池）
        pipeline.addLast(businessGroup, "clientHandler",
            new ClientHandler(handlerChain, sessionManager));
        
        // 4. 重连处理器
        pipeline.addLast("reconnectHandler", 
            new ReconnectHandler(this));
        
        // 5. 心跳处理器
        pipeline.addLast("heartbeatHandler",
            new HeartbeatHandler(this));
        
        // 6. 日志处理器（可选）
        if (config.isEnableLog()) {
            pipeline.addLast("logging",
                new LoggingHandler(LogLevel.DEBUG));
        }
    }
    
    /**
     * 连接服务器
     */
    public void connect() {
        if (clientChannel != null && clientChannel.isActive()) {
            log.info("Client is already connected");
            return;
        }
        
        bootstrap.connect(config.getServerIp(), config.getServerPort())
            .addListener((ChannelFutureListener) future -> {
                if (future.isSuccess()) {
                    clientChannel = future.channel();
                    log.info("Connected to server: {}:{}", 
                        config.getServerIp(), config.getServerPort());
                    
                    // 发送登录请求
                    login();
                    
                    // 启动心跳
                    startHeartbeat();
                    
                } else {
                    log.error("Failed to connect to server: {}:{}",
                        config.getServerIp(), config.getServerPort());
                    
                    // 触发重连
                    if (config.getReconnect().isEnabled()) {
                        scheduleReconnect();
                    }
                }
            });
    }
    
    /**
     * 断开连接
     */
    @Override
    public void stop() {
        if (!running) {
            return;
        }
        
        log.info("Stopping JT809 Client...");
        
        // 1. 停止心跳和重连任务
        stopHeartbeat();
        stopReconnect();
        
        // 2. 发送断开请求
        if (clientChannel != null && clientChannel.isActive()) {
            sendDisconnectRequest();
        }
        
        // 3. 关闭Channel
        if (clientChannel != null) {
            clientChannel.close().awaitUninterruptibly();
        }
        
        // 4. 优雅关闭EventLoopGroup
        shutdown();
        
        running = false;
        log.info("JT809 Client stopped");
    }
    
    /**
     * 发送消息
     */
    public void send(BaseMessage message) {
        if (clientChannel != null && clientChannel.isActive()) {
            clientChannel.writeAndFlush(message).addListener((ChannelFutureListener) future -> {
                if (!future.isSuccess()) {
                    log.error("Send message failed", future.cause());
                }
            });
        } else {
            log.warn("Client channel is not active, cannot send message");
        }
    }
    
    /**
     * 登录
     */
    private void login() {
        UpConnectReq loginReq = new UpConnectReq();
        UpConnectReq.Body body = new UpConnectReq.Body();
        body.setUserId(config.getPlatformId());
        body.setPassword(config.getPassword());
        body.setDownLinkIp(config.getLocalIp());
        body.setDownLinkPort(config.getLocalPort());
        loginReq.setBody(body);
        
        send(loginReq);
        log.info("Sent login request to server");
    }
    
    /**
     * 发送断开请求
     */
    private void sendDisconnectRequest() {
        UpDisconnectReq disconnectReq = new UpDisconnectReq();
        send(disconnectReq);
        log.info("Sent disconnect request to server");
    }
    
    /**
     * 启动心跳
     */
    private void startHeartbeat() {
        if (!config.getHeartbeat().isEnabled()) {
            return;
        }
        
        stopHeartbeat();
        
        heartbeatFuture = workerGroup.scheduleAtFixedRate(() -> {
            if (clientChannel != null && clientChannel.isActive() && loginSuccess) {
                UpLinkTestReq heartbeat = new UpLinkTestReq();
                send(heartbeat);
                log.debug("Sent heartbeat to server");
            }
        }, config.getHeartbeat().getInterval(), 
           config.getHeartbeat().getInterval(), 
           TimeUnit.MILLISECONDS);
        
        log.info("Started heartbeat task, interval: {}ms", 
            config.getHeartbeat().getInterval());
    }
    
    /**
     * 停止心跳
     */
    private void stopHeartbeat() {
        if (heartbeatFuture != null) {
            heartbeatFuture.cancel(false);
            heartbeatFuture = null;
        }
    }
    
    /**
     * 调度重连
     */
    public void scheduleReconnect() {
        if (!config.getReconnect().isEnabled()) {
            return;
        }
        
        stopReconnect();
        
        reconnectFuture = workerGroup.schedule(() -> {
            log.info("Reconnecting to server: {}:{} ...", 
                config.getServerIp(), config.getServerPort());
            connect();
        }, config.getReconnect().getInterval(), TimeUnit.MILLISECONDS);
        
        log.info("Scheduled reconnect after {}ms", 
            config.getReconnect().getInterval());
    }
    
    /**
     * 停止重连
     */
    private void stopReconnect() {
        if (reconnectFuture != null) {
            reconnectFuture.cancel(false);
            reconnectFuture = null;
        }
    }
    
    /**
     * 优雅关闭
     */
    private void shutdown() {
        if (workerGroup != null) {
            workerGroup.shutdownGracefully(100, 300, TimeUnit.MILLISECONDS);
        }
        if (businessGroup != null) {
            businessGroup.shutdownGracefully(100, 300, TimeUnit.MILLISECONDS);
        }
    }
    
    @Override
    public boolean isRunning() {
        return running;
    }
    
    @Override
    public int getPhase() {
        return 0;
    }
    
    @Override
    public boolean isAutoStartup() {
        return true;
    }
    
    @Override
    public void stop(Runnable callback) {
        stop();
        callback.run();
    }
    
    // Getter
    public boolean isLoginSuccess() {
        return loginSuccess;
    }
    
    public void setLoginSuccess(boolean loginSuccess) {
        this.loginSuccess = loginSuccess;
    }
}
```

#### 3.6.2 处理器链
```java
/**
 * 消息处理器链
 */
public class HandlerChain {
    private final List<MessageHandler> handlers = new ArrayList<>();
    
    public void addHandler(MessageHandler handler) {
        handlers.add(handler);
        handlers.sort(Comparator.comparingInt(MessageHandler::getOrder));
    }
    
    public void process(MessageContext context, BaseMessage message) {
        for (MessageHandler handler : handlers) {
            if (Arrays.binarySearch(handler.supportedMessageTypes(), 
                message.getHead().getMsgId()) >= 0) {
                if (!handler.handle(context, message)) {
                    break;
                }
            }
        }
    }
}
```

## 8. 关键技术点

### 8.1 消息转义处理
JT809协议要求对特定字节进行转义：
- `5b` → `5a 01`
- `5a` → `5a 02`
- `5e` → `5e 01`

### 8.2 CRC校验
使用CRC-CCITT (0xFFFF)算法进行校验：

```java
public class CRCUtil {
    public static int calculate(byte[] data) {
        int crc = 0xFFFF;
        for (byte b : data) {
            crc ^= (b & 0xFF);
            for (int i = 0; i < 8; i++) {
                if ((crc & 0x0001) != 0) {
                    crc >>= 1;
                    crc ^= 0xA001;
                } else {
                    crc >>= 1;
                }
            }
        }
        return crc;
    }
}
```

### 9.2 零拷贝优化

#### 9.2.1 使用CompositeByteBuf
```java
/**
 * 消息组装优化 - 使用CompositeByteBuf避免内存拷贝
 */
public class MessageAssembler {
    
    /**
     * 传统方式：多次内存拷贝
     */
    public byte[] assembleMessageTraditional(MessageHead head, MessageBody body) {
        byte[] headBytes = encodeHead(head);
        byte[] bodyBytes = encodeBody(body);
        
        // 创建新数组并拷贝，产生额外内存开销
        byte[] result = new byte[headBytes.length + bodyBytes.length];
        System.arraycopy(headBytes, 0, result, 0, headBytes.length);
        System.arraycopy(bodyBytes, 0, result, headBytes.length, bodyBytes.length);
        
        return result;
    }
    
    /**
     * 优化方式：零拷贝
     */
    public ByteBuf assembleMessageZeroCopy(MessageHead head, MessageBody body) {
        ByteBuf headBuf = encodeHeadToByteBuf(head);
        ByteBuf bodyBuf = encodeBodyToByteBuf(body);
        
        // 使用CompositeByteBuf逻辑组合，不产生实际拷贝
        CompositeByteBuf composite = ByteBufManager.compositeBuffer();
        composite.addComponents(true, headBuf, bodyBuf);
        
        return composite;
    }
}
```

```java
            @Override
            protected PooledMessage newObject(Handle<PooledMessage> handle) {
                return new PooledMessage(handle);
            }
        };
    
    /**
     * 从池中获取对象
     */
    public static PooledMessage newInstance() {
        return RECYCLER.get();
    }
```

```java
/**
 * Netty线程工厂配置
 */
public class NettyThreadFactory implements ThreadFactory {
    
    private final String namePrefix;
    private final AtomicInteger threadNumber = new AtomicInteger(1);
    private final boolean daemon;
    
    public NettyThreadFactory(String namePrefix, boolean daemon) {
        this.namePrefix = namePrefix;
        this.daemon = daemon;
    }
    
    @Override
    public Thread newThread(Runnable r) {
        Thread thread = new Thread(r, namePrefix + "-" + threadNumber.getAndIncrement());
        thread.setDaemon(daemon);
        
        // 设置优先级
        if (thread.getPriority() != Thread.NORM_PRIORITY) {

        // 设置未捕获异常处理器
        thread.setUncaughtExceptionHandler((t, e) -> {
            log.error("Uncaught exception in thread: {}", t.getName(), e);
        });
        
        return thread;
    }
}
```

### 9.6 批量处理优化

#### 9.6.1 消息批量发送
```java
/**
 * 批量消息发送优化
 * 使用Netty的批量写操作
 */
public class BatchMessageSender {
    
    private final Channel channel;
    
    /**
     * 批量发送消息
     * 避免多次writeAndFlush调用
     */
    public void sendBatch(List<BaseMessage> messages) {
        if (channel == null || !channel.isActive()) {
            return;
        }
        
        // 开始批量写
        channel.write(messages.get(0));
        
        for (int i = 1; i < messages.size(); i++) {
            channel.write(messages.get(i));
        }
        
        // 一次性flush
        channel.flush();
    }
    
    /**
     * 使用CompositeByteBuf批量组装消息
     */
    public void sendBatchOptimized(List<BaseMessage> messages) {
        CompositeByteBuf composite = ByteBufManager.compositeBuffer();
        
        for (BaseMessage message : messages) {
            ByteBuf encoded = encodeMessage(message);
            composite.addComponent(true, encoded);
        }
        
        channel.writeAndFlush(composite);
    }
}
```

```java
/**
 * 高频消息批量处理器
 * 使用队列缓冲 + 定时批量处理
 */
@Component
public class HighFrequencyMessageHandler {
    
    private final Queue<BaseMessage> messageQueue = new LinkedBlockingQueue<>(10000);
    
    @Autowired
    private MessageProcessor messageProcessor;
    
    /**
     * 接收消息到队列
     */
    public void receive(BaseMessage message) {
        if (!messageQueue.offer(message)) {
            log.warn("Message queue is full, dropping message");
        }
    }
    
    /**
     * 定时批量处理（每100ms执行一次）
     */
    @Scheduled(fixedRate = 100)
    public void processBatch() {
        List<BaseMessage> batch = new ArrayList<>(100);
        messageQueue.drainTo(batch, 100);
        
        if (!batch.isEmpty()) {
            // 批量处理
            messageProcessor.processBatch(batch);
        }
    }
}
```
